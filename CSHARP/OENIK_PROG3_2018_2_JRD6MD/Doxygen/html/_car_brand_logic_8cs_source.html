237  public IQueryable CarsFullPrice()
238  {
239  // Legyen lehetőségünk kiírni minden autóhoz az autó TELJES árát is: alapár + a rajta lévő extrák összára
240  var carbrands = this.carBrandRepository.ReadAll();
241  var extras = this.extraRepository.ReadAll();
242  var models = this.modelRepository.ReadAll();
243  var modelextraswitch = this.modelExtraSwitchRepository.ReadAll();
244 
245  var prepreResult = modelextraswitch.Join(models, x => x.Model_Id, y => y.Model_Id, (e, d) => new
246  {
247  MODELID = d.Model_Id,
248  MODELNAME = d.Model_Name,
249  EXTRAID = e.Extra_Id,
250  MODELPRICE = d.Model_Base_Price
251  }).Join(extras, z => z.EXTRAID, c => c.Extra_Id, (a, b) => new
252  {
253  MODELID = a.MODELID,
254  MODELNAME = a.MODELNAME,
255  EXTRAID = a.EXTRAID,
256  FULLPRICE = a.MODELPRICE + b.Extra_Price
257  }).GroupBy(x => x.MODELNAME);
258 
259  var preresult = prepreResult.Select(x => new
260  {
261  MODELNAME = x.Key,
262  Price = x.Sum(y => y.FULLPRICE)
263  });
264 
265  var result = preresult.Select(x => "Model name: " + x.MODELNAME + "\n\t Price with all extra on it: " + x.Price + " euro");
266  return result;
267  }
